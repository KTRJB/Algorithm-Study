### 예톤

```swift
func fibo(_ n: Int) -> Int {
    var cache: [Int] = [0, 1]
    
    for num in 2...n {
        cache.append(cache[num - 1] + cache[num - 2])
    }
    
    return cache[n]
}

fibo(4)
```

### Judy
```swift
func fib(_ n: Int) -> Int {
    if n == 0 {
        return 0
    } else if n == 1 {
        return 1
    } else {
        return fib(n-1) + fib(n-2)
    }
}

// DP
func fib(_ n: Int) -> Int {
    var m: [Int: Int] = [0: 0, 1: 1]
    
    if m[n] == nil {
        m[n] = fib(n-1) + fib(n-2)
    }
    
    return m[n]!
}
```
